rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow read access to all authenticated users
    match /{document=**} {
      allow read: if request.auth != null;
    }
    
    // Admin-only write access
    match /donors/{donor} {
      allow write: if request.auth != null && 
        exists(/databases/$(database)/documents/admins/$(request.auth.email));
    }
    
    match /requests/{request} {
      allow write: if request.auth != null && 
        exists(/databases/$(database)/documents/admins/$(request.auth.email));
    }
    
    match /inventory/{item} {
      allow write: if request.auth != null && 
        exists(/databases/$(database)/documents/admins/$(request.auth.email));
    }
    
    match /events/{event} {
      allow write: if request.auth != null && 
        exists(/databases/$(database)/documents/admins/$(request.auth.email));
    }
    
    match /admins/{email} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && 
        request.auth.email == email || // Allow users to write their own admin doc
        exists(/databases/$(database)/documents/admins/$(request.auth.email)); // Or if they're already an admin
    }
    
    match /site_settings/{setting} {
      allow read: if true;
      allow write: if request.auth != null && 
        exists(/databases/$(database)/documents/admins/$(request.auth.email));
    }
  }
}
